```sql
CREATE VIEW STATISTIKA_RADNIH_MESTA
AS
SELECT
  Z.IME,
  Z.PREZIME,
  COUNT(DISTINCT IRM.ID_R_MESTO) AS BROJ_RADNIH_MESTA,
  SUM(IRM.DATUM_KRAJA - IRM.DATUM_POC) AS RADNI_DANI
FROM
  ZAPOSLENI Z
INNER JOIN
  ISTORIJA_RADNIH_MESTA IRM
ON
  Z.ID = IRM.ID_ZAP
GROUP BY
  Z.IME,
  Z.PREZIME;

SELECT *
FROM
  STATISTIKA_RADNIH_MESTA
WHERE
  RADNI_DANI > 100
AND
  BROJ_RADNIH_MESTA >= 2;
```
