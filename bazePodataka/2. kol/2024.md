```sql
CREATE VIEW STATISTIKA_ZAPOSLENI
AS
SELECT
  Z.IME,
  Z.PREZIME,
  COUNT(*) AS BROJ_POZICIJA,
  AVG(RM.DATUMDO - RM.DATUMOD) AS VREME
FROM
  ZAPOSLENI Z
INNER JOIN
  RADNO_MESTO RM
ON
  Z.JMBG = RM.ZJMBG
WHERE
  RM.DATUMDO IS NOT NULL
GROUP BY
  Z.IME,
  Z.PREZIME;

SELECT *
FROM
  STATISTIKA_ZAPOSLENI
WHERE
  BROJ_POZICIJA > 1
AND
  VREME > 300;
```
